Marionette Sortable Collection View

Sample Usage

  // Single item view to display
  var SortableLi = Marionette.SortableItemView.extend({
    template: _.template('<img src="<%=posters.thumbnail%>" alt="" /> %lt;h3 class="inliner"><%=title%>'),
    className: 'box'
  });

  // Collection View
  var SortableList = Marionette.SortableCollectionView.extend({
    itemView: Views.SortableLi
  });

  // Get some Data
  var collection = new Backbone.Collection();
  $.getJSON('http://api.rottentomatoes.com/api/public/v1.0/lists/movies/box_office.json?apikey=[your-api-key]&limit=50&callback=?')
    .done(function(data) {
      collection.reset(data.movies);
    });

  // Create the view and assign the collection to it
  var view = new Views.SortableList({
    collection: collection
  });

  // Show it in a region (in this case content)
  this.content.show(view);
